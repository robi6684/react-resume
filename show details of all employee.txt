Querries on Region-Country-location database->

1. Show all locations of a country having country id as "IT"
SELECT * FROM LOCATIONS WHERE COUNTRY_ID = "IT";

2. Show all location names country wise in a region "Asia"
SELECT LOCATIONS.CITY FROM REGIONS INNER JOIN 
COUNTRIES ON REGIONS.REGION_ID = COUNTRIES.REGION_ID
INNER JOIN LOCATIONS ON COUNTRIES.COUNTRY_ID = LOCATIONS.COUNTRY_ID
WHERE REGIONS.REGION_NAME = "ASIA";

3. Show all postal codes of United States
SELECT LOCATIONS.POSTAL_CODE FROM LOCATIONS INNER JOIN COUNTRIES ON 
LOCATIONS.COUNTRY_ID = COUNTRIES.COUNTRY_ID WHERE COUNTRY_NAME = "USA";

4. Show number of countries in each region
SELECT R.REGION_NAME, COUNT(C.COUNTRY_ID) AS COUNTRY_COUNT 
FROM REGIONS R INNER JOIN COUNTRIES C ON R.REGION_ID = C.REGION_ID GROUP BY R.REGION_NAME;

5. Show postal codes of locations country wise ignoring locations not having state province 
SELECT L.POSTAL_CODE, C.COUNTRY_NAME
FROM COUNTRIES C JOIN LOCATIONS L ON C.COUNTRY_ID = L.COUNTRY_ID
WHERE L.STATE_PROVINCE IS NOT NULL;

6. Show names of countries in each region in descending order of number of locations in that country
SELECT R.REGION_NAME, C.COUNTRY_NAME, COUNT(L.LOCATION_ID) AS LOCATIONS_COUNT
FROM REGIONS R JOIN COUNTRIES C ON R.REGION_ID = C.REGION_ID
JOIN LOCATIONS L ON C.COUNTRY_ID = L.COUNTRY_ID
GROUP BY R.REGION_NAME, C.COUNTRY_NAME ORDER BY LOCATIONS_COUNT DESC;

7. Show countries having maximum locations with number of locations in it
SELECT COUNTRIES.COUNTRY_NAME, COUNT(LOCATIONS.LOCATION_ID) AS LOCATIONS_COUNT
FROM LOCATIONS INNER JOIN COUNTRIES ON LOCATIONS.COUNTRY_ID = COUNTRIES.COUNTRY_ID
GROUP BY COUNTRIES.COUNTRY_NAME ORDER BY LOCATIONS_COUNT DESC LIMIT 1;

8. Count number of locations in each region
SELECT REGIONS.REGION_NAME, COUNT(LOCATIONS.LOCATION_ID) AS LOCATIONS_COUNT
FROM REGIONS INNER JOIN COUNTRIES ON REGIONS.REGION_ID = COUNTRIES.REGION_ID
INNER JOIN LOCATIONS ON COUNTRIES.COUNTRY_ID = LOCATIONS.COUNTRY_ID GROUP BY 
REGIONS.REGION_NAME;
